#### 1.DFS
```cpp
const int N = 20;
int map[N][N] = { 0 };
bool visited[N][N] = { false };

int dx[4] = {};
int dy[4] = {};
int n, k;
string ans;

void dfs(int x, int y,int depth, string s) {
	if (x == n && y == n && depth == n * n) {
		ans = ans < s ? ans : s;
		return;
	}
	for (int i = 0; i < 4; i++) {
		int bx = x + dx[i], by = y + dy[i];
		if (bx < 1 || bx > n || by < 1 || by > n) continue;
		if (visited[bx][by]) continue;

		visited[bx][by] = true;
		dfs(bx, by, depth + 1, s + to_string(i));
		visited[bx][by] = false;
	}
}
```
