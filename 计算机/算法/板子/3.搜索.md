#### 1.DFS
```cpp
const int N = 20;
int map[N][N] = { 0 };
bool visited[N][N] = { false };

int dx[8] = {};
int dy[8] = {};
int n, k;
string ans;

void dfs(int x, int y,int depth, string s) {
	if (x == n && y == n && depth == n * n) {
		ans = ans < s ? ans : s;
		return;
	}
	for (int i = 0; i < 4; i++) {
		int bx = x + dx[i], by = y + dy[i];
		if (bx < 1 || bx > n || by < 1 || by > n) continue;
		if (visited[bx][by]) continue;

		visited[bx][by] = true;
		dfs(bx, by, depth + 1, s + to_string(i));
		visited[bx][by] = false;
	}
}
```

#### 2.BFS
```cpp
const int N = 1001;
const int M = 1001;
char brr[N][M] = {'#'};
bool vis[N][M][11] = {false};
int dx[4] = {1,0,-1,0};
int dy[4] = {0,1,0,-1};

struct node {
	int x,y,cnt,step;
	node(int y,int x,int cnt,int step):y(y),x(x),cnt(cnt),step(step) {
	};
};

int n,m,k;
int ans = INT_MAX;
bool isA = true;

void bfs(int y,int x) {
	queue<node> q;
	q.push(node(y,x,0,1));

	while(!q.empty()) {
		node nd = q.front();
		q.pop();
		int ny = nd.y ,nx = nd.x;
		if(ny == n  && nx == n ) {
			ans = min(ans,nd.step - 1);
			continue;
		}
		char tag =((nd.step  / k) % 2 == 0 ? 'A' : 'B');
		// cout << ny << " " << nx << " " << tag << endl;
		for(int i = 0 ; i < 4; i++) {
			int by = ny + dy[i] ,bx = nx + dx[i], bcnt = (nd.cnt+1) % k;
			if(brr[by][bx] != tag) continue;
			if(vis[by][bx][bcnt]) continue;

			// cout << by << " " << bx << " " << bcnt << endl;
			vis[by][bx][bcnt] = true;
			q.push(node(by,bx,bcnt, nd.step + 1));
		}
	}
}
```
